import common from '../common.js'
export default {
  data () {
    return {
      activityTerm: null,
      termList: null,
      page: null,
      activityPage: 1,
      prevPage: null,
      nextPage: null,
      activityId: null,
      activityStatus: null,
      statusList: null,
      studentMsg: null,
      applyList: null,
      geted: false,
      geted2: false
    }
  },
  methods: {
    doneThing: function (data) {
      console.log(data)
      if (data.status) {
        this.getApplyList()
      };
      common.alertBox(data.message)
    },
    getTermList () {
      common.ajax('get', '/info/idcards/getclass', this.setTermList)
    },
    setTermList (data) {
      console.log(data)
      if (data.status) {
        this.termList = data.result
        this.activityTerm = this.termList[0].dm
        this.getPageList()
      } else {
        common.alertBox(data.message)
      }
    },
    getPageList () {
      common.ajax('get', '/info/idcards/management/' + this.activityTerm, this.setPage)
    },
    setPage (data) {
      console.log(data)
      if (data.status) {
        this.page = data.result[0].ys
        this.getApplyList()
      } else {
        common.alertBox(data.message)
      }
    },
    getStatusList () {
      common.ajax('get', '/info/idcards/managements/status', this.setStatusList)
    },
    setStatusList (data) {
      console.log(data)
      if (data.status) {
        this.statusList = data.result
      }
    },
    getApplyList (add) {
      if (add === '+') {
        this.activityPage = this.nextPage
      } else if (add === '-') {
        this.activityPage = this.prevPage
      }
      this.prevPage = this.activityPage - 1
      this.nextPage = this.activityPage + 1
      common.ajax('get', '/info/idcards/management/' + this.activityTerm + '/' + this.activityPage, this.setApplyList)
    },
    setApplyList (data) {
      console.log(data)
      if (data.status) {
        this.applyList = data.result
        this.geted = true
      } else {
        common.alertBox(data.message)
      }
    },
    getedStudentMsg (event) {
      this.geted2 = false
      let cid = event.currentTarget.getAttribute('cid')
      common.ajax('get', '/info/idcards/user/' + cid, this.setStudentMsg)
    },
    setStudentMsg (data) {
      console.log(data)
      if (data.status) {
        this.studentMsg = data.infos[0]
        this.geted2 = true
      } else {
        common.alertBox(data.message)
      }
    },
    changeActivityId (event) {
      this.activityId = event.currentTarget.getAttribute('cid')
    },
    manage () {
      let jsonData = JSON.stringify({
        sqzt_code: this.activityStatus
      })
      common.ajax('put', '/info/idcards/managements/apply/' + this.activityId, this.doneThing, jsonData, 'json')
    }
  },
  ready () {
    this.getTermList()
    // this.getStatusList()
  }
}
