import common from '../common.js'
export default {
  data () {
    return {
      collegeList: null,
      majorList: null,
      activityCollege: null,
      addMajor: {
        zyh: null,
        zymc: null,
        xy_code: null,
        ksnf: null,
        xz: null
      },
      editMajor: {
        zymc: null,
        xy_code: null,
        ksnf: null,
        xz: null
      },
      activityMajor: null,
      oneMajorMsg: null
    }
  },
  methods: {
    doneThing: function (data) {
      console.log(data)
      if (data.status) {
        this.getMajorList()
        common.alertBox(data.message)
      } else {
        common.alertBox(data.message)
      }
    },
    activityOneMajor: function (event) {
      let cid = event.currentTarget.getAttribute('cid')
      this.activityMajor = cid
    },
    getCollegeList: function () {
      if (localStorage.getItem('college')) {
        this.collegeList = JSON.parse(localStorage.getItem('college'))
        this.activityCollege = this.collegeList[0].yxsh
        this.addMajor.xy_code = this.collegeList[0].yxsh
        console.log(this.activityCollege)
        this.getMajorList()
        console.log(this.collegeList)
      } else {
        common.ajax('get', '/info/colleges', this.setCollegeList, {}, 'json')
      }
    },
    setCollegeList: function (data) {
      console.log(data)
      if (data.status) {
        this.collegeList = data.colleges
        localStorage.setItem('college', JSON.stringify(data.colleges))
        this.activityCollege = this.collegeList[0].yxsh
        this.addMajor.xy_code = this.collegeList[0].yxsh
        this.getMajorList()
      } else {
        common.alertBox(data.message)
      }
    },
    getMajorList: function () {
      let data = {
        xy_code: this.activityCollege
      }
      let jsonData = JSON.stringify(data)
      common.ajax('post', '/info/major_list', this.setMajorList, jsonData, 'json')
    },
    setMajorList: function (data) {
      console.log(data)
      if (data.status) {
        this.majorList = data.majors
      } else {
        common.alertBox(data.message)
      }
    },
    getOneMajor: function (event) {
      this.activityOneMajor(event)
      common.ajax('get', '/info/major/' + this.activityMajor, this.setOneMajor, {}, 'json')
    },
    setOneMajor: function (data) {
      console.log(data)
      if (data.status) {
        this.oneMajorMsg = data.major
      } else {
        common.alertBox(data.message)
      };
    },
    addOneMajor: function () {
      let jsonData = JSON.stringify(this.addMajor)
      common.ajax('post', '/info/majors', this.doneThing, jsonData, 'json')
    },
    editOneMajor: function () {
      let jsonData = JSON.stringify(this.editMajor)
      common.ajax('put', '/info/major/' + this.activityMajor, this.doneThing, jsonData, 'json')
    }
  },
  ready () {
    this.getCollegeList()
  }
}
