                firCode: '请选择',

                activityFirCode: null,
                activitySecCode: null,
                activityThrCode: null,

                parentMsg: {
                    fir: null,
                    sec: null,
                    thr: null
                },
                isSet: true
            }
        },
        methods: {
            getFirCode: function() {
                let data = {
                    level: 1,
                    code: null
                };
                let jsonData = JSON.stringify(data);
                common.ajax('post', '/info/code_xzqh', this.setFirCode, jsonData, 'json');
            },
            getSecCode: function() {
                let data = {
                    level: 2,
                    code: this.activityFirCode
                };
                let jsonData = JSON.stringify(data);
                common.ajax('post', '/info/code_xzqh', this.setSecCode, jsonData, 'json');
            },
            getThrCode: function() {
                let data = {
                    level: 3,
                    code: this.activitySecCode
                };
                let jsonData = JSON.stringify(data);
                common.ajax('post', '/info/code_xzqh', this.setThrCode, jsonData, 'json');

                // console.log(data);
                if (data.status) {
                    this.firCode = data.code;
                    this.parentMsg.fir = this.parentMsg.fir
                } else {
                    common.alertBox(data.message);
                };
            },
            setSecCode: function(data) {
                // console.log(data);
                if (data.status) {
                    this.secCode = data.code;
                    if (this.isSet) {
                        this.activitySecCode = this.parentMsg.sec;
                    }
                    this.activitySecCode = this.secCode[1].dm

                } else {
                    common.alertBox(data.message);
                };

                // console.log(data);

                    this.thrCode = data.code;
                    if (this.isSet) {
                    this.parentMsg.thr = this.parcode;
                    }
                    this.parentMsg.thr = this.thrCode[1].dm;
                } else {
                    common.alertBox(data.message);
                };
                this.isSet = false
            },
