import common from '../../common.js';
export default {
    data() {
            return {
                secCode: null,
                activityFirCode: null,
                firCode: null,
                activitySecCode: null,
                parcode: null,
                parentMsg: {
                    fir: null,
                    sec: null
                },
                isSet:true
            }
        },
        methods: {
            getFirCode: function() {
                let data = {
                    level: 1,
                    code: null
                };
                let jsonData = JSON.stringify(data);
                common.ajax('post', '/info/code_zcjb', this.setFirCode, jsonData, 'json');
            },
            setFirCode: function(data) {
                console.log(data);
                if (data.status) {
                    this.firCode = data.code;
                     this.activityFirCode = this.parentMsg.fir == null ?this.firCode[0].dm:this.parentMsg.fir
                    this.getSecCode()
                } else {
                    common.alertBox(data.message);
                };
            },
            getSecCode: function() {
                let data = {
                    level: 2,
                    code: this.activityFirCode
                };
                let jsonData = JSON.stringify(data);
                common.ajax('post', '/info/code_zcjb', this.setSecCode, jsonData, 'json');
            },
            setSecCode: function(data) {
                console.log(data);
                if (data.status) {
                    this.secCode = data.code;
                     if (this.isSet) {
                        this.parentMsg.sec = this.parcode;
                        this.activitySecCode = this.parcode
                    }else{
                        this.parentMsg.sec = this.secCode[0].dm
                        this.activitySecCode = this.secCode[0].dm
                    }
                    this.isSet = false
                } else {
                    common.alertBox(data.message);
                };
            },
        },
        events: {
            'getde-msg': function(msg) {
                 this.getFirCode();
                console.log(msg);
                if (msg.zcjb.dm) {
                    this.parcode = parseInt(msg.zcjb.dm);
                    let code = this.parcode.split[''];
                    console.log(code);
                    this.parentMsg.fir = code[0]+'00';
                    this.activityFirCode = this.parentMsg.fir;
                }
            }
