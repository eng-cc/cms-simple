import common from '../../common.js'
import jgBar from './jgBar.vue'
export default {
    data() {
            return {
                xb: JSON.parse(localStorage.getItem('xb')),
                xslb: JSON.parse(localStorage.getItem('xslb')),
                mz: JSON.parse(localStorage.getItem('mz')),
                gb: JSON.parse(localStorage.getItem('gb')),
                pyfs: JSON.parse(localStorage.getItem('pyfs')),
                zzmm: JSON.parse(localStorage.getItem('zzmm')),
                oneStudentMsg: {
                    basic: null,
                    roll: null,
                    family: null
                },
                xh: null,
                editMsg: {
                    basic: {
                        sjh:null,
                        qqh:null,
                        jhkh:null,
                        qsh:null
                    },
                    roll: null,
                    family: null
                },
                showAll: 0
            }
        },
        components: {
            jgBar: jgBar
        },
        methods: {
            doneThing: function(data) {
                console.log(data)
                if (data.status) {
                    this.getStudentMsg();
                };
                common.alertBox(data.message)
            },
            getStudentMsg: function(event) {
                this.showAll = 0
                common.ajax('get', '/info/student/basic/' + this.xh, this.setStudentMsgBasic)
                common.ajax('get', '/info/student/school_roll/' + this.xh, this.setStudentMsgRoll)
                common.ajax('get', '/info/student/family/' + this.xh, this.setStudentMsgFamily)
            },
            setStudentMsgBasic: function(data) {
                console.log(data)
                if (data.status) {
                    this.oneStudentMsg.basic = data.basic_info
                    this.$broadcast('jgBar-msg', this.oneStudentMsg.basic)
                    this.showAll += 1
                } else {
                    common.alertBox(data.message)
                }
            },
            setStudentMsgRoll: function(data) {
                console.log(data)
                if (data.status) {
                    this.oneStudentMsg.roll = data.school_roll.reverse()
                    this.showAll += 1
                } else {
                    common.alertBox(data.message)
                }
            },
            setStudentMsgFamily: function(data) {
                console.log(data)
                if (data.status) {
                    this.oneStudentMsg.family = data.family
                    this.$broadcast('family-msg', this.oneStudentMsg.family)
                    this.showAll += 1
                } else {
                    common.alertBox(data.message)
                }
            },
            editBasicBase: function() {
                let jsonData = JSON.stringify(this.editMsg.basic)
                common.ajax('put','/info/student/contact/'+this.xh,this.doneThing,jsonData,'json')
            }
        },
        events: {
            'xh-msg': function(msg) {
                console.log(msg)
                this.xh = msg
                this.getStudentMsg()
            }
        },
        ready() {
 
        }
}
