    getApplyList () {
      this.geted = false
      this.activityTerm = this.xueqi[0].dm
      common.ajax('get', '/supp/spark/count/' + this.activityTerm, this.setApplyList)
    },
    setApplyList (data) {
      console.log(data)
      if (data.status) {
        this.applyList = data.count
        this.xsCount = 0
        this.sqCount = 0
        this.shCount = 0
        for (let i = data.count.length - 1; i >= 0; i--) {
          data.count[i].xs = this.xsrs[i]
          this.xsCount += data.count[i].xs
          this.sqCount += data.count[i].xhsq
          this.shCount += data.count[i].xhsh
          this.geted = true
        }
      } else {
        common.alertBox(data.meassage)
      }
    }
  },
  ready () {
    this.getApplyList()
