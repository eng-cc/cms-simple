import common from '../common.js'
export default {
  data () {
    return {
      opinionList: null,
      geted: false,
      oneOpinion: null,
      backMsg: {
        content: null
      },
      activityId: null,
      activityTitle: null,
      activityName: null,
      pages: null,
      prevNum: null,
      nextNum: null,
      activityPage: null
    }
  },
  methods: {
    doneThing: function (data) {
      console.log(data)
      if (data.status) {
        this.getOpinionList(this.activityPage)
      };
      common.alertBox(data.message)
    },
    toActivityId (event) {
      this.activityId = event.currentTarget.getAttribute('cid')
      this.activityTitle = event.currentTarget.getAttribute('data-title')
      this.activityName = event.currentTarget.getAttribute('data-name')
    },
    getOneOpinion (event) {
      this.toActivityId(event)
      let cid = event.currentTarget.getAttribute('cid')
      common.ajax('get', '/feedback/detail/' + cid, this.setOneOpinion)
    },
    setOneOpinion (data) {
      console.log(data)
      if (data.status) {
        this.oneOpinion = data.data
      } else {
        common.alertBox(data.message)
      }
    },
    getOpinionList (num) {
      this.activityPage = num
      this.geted = false
      common.ajax('get', '/feedback/admin/list/' + num, this.setOpinionList)
    },
    setOpinionList (data) {
      console.log(data)
      if (data.status) {
        this.opinionList = data.data
        this.geted = true
        this.prevNum = data.prev
        this.nextNum = data.next
        this.pages = data.pages
      } else {
        common.alertBox(data.message)
      }
    },
    backOneOpinion () {
      let jsonData = JSON.stringify(this.backMsg)
      common.ajax('post', '/feedback/admin/' + this.activityId, this.doneThing, jsonData, 'json')
    },
    deleteOpinion () {
      let data = {
        id: this.activityId
      }
      let jsonData = JSON.stringify(data)
      common.ajax('delete', '/feedback/admin/' + this.activityId, this.doneThing, jsonData, 'json')
    },
    sendMsm () {
      common.ajax('put', '/feedback/admin/' + this.activityId, this.doneThing)
    }
  },
  ready () {
    this.getOpinionList(1)
  }
}
