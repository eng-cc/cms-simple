import common from '../common.js'
var wait = 60
export default {
  data () {
    return {
      username: null,
      phonenumber: null,
      testcode: null,
      password: null,
      repassword: null
    }
  },
  methods: {
    getPhoneNumber: function () {
      let oldusername = localStorage.getItem('username')
      if (this.username.length > 4 && oldusername !== this.username) {
        let url = '/user/reset_password?id=' + this.username
        let token = common.tokenOperate.takeToken()
        let header = [{
          k: 'Authorization',
          v: 'passport ' + token
        }]
        localStorage.setItem('username', this.username)
        localStorage.removeItem('username')
        common.ajax('get', url, this.usePhoneNumber, {}, 'json', header)
      }
    },
    usePhoneNumber: function (data) {
      if (data.status) {
        this.phonenumber = data.phone_num
      } else {
        alert(data.message) // eslint-disable-line
      };
    },
    getTestcode: function () {
      let data = {
        school_id: this.username
      }
      let jsonData = JSON.stringify(data)
      common.ajax('put', '/user/reset_password', this.useTestcode, jsonData, 'json')
    },
    testcodeTime: function (event) {
      let el = event.currentTarget
      let temout = (el) => {
        if (wait === 0) {
          el.removeAttribute('disabled')
          el.value = '获取验证码'
          wait = 60
        } else {
          el.setAttribute('disabled', true)
          el.value = '重新发送(' + wait + ')'
          wait--; // eslint-disable-line
          setTimeout(function () {
            temout(el)
          },
            1000)
        }
      }
      temout(el)
      this.getTestcode()
    },
    useTestcode: (data) => {
      if (data.status) {
        common.alertBox('发送成功')
      } else {
        common.alertBox('发送失败')
      }
    },
    resetPassword: function () {
      if (this.password === this.repassword) {
        let data = {
          verify_code: this.testcode,
          password: this.password,
          school_id: this.username
        }
        let jsondata = JSON.stringify(data)
        common.ajax('post', '/user/reset_password', this.succRePassword, jsondata, 'json')
      } else {
        common.alertBox('密码不一致')
      }
    },
    succRePassword: function (data) {
      if (data.status) {
        common.alertBox('修改成功')
        router.go('/index/indexApp')
      } else {
        common.alertBox(data.message)
      }
    }
  }
}
