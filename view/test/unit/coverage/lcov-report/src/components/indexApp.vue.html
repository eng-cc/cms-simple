      appTypes: null,
      done: false
    }
  },
  methods: {
    getAppTypes: function () {
      common.ajax('get', '/admin/app_types', this.setAppTypes, {}, 'json')
    },
    setAppTypes: function (data) {
      console.log(data)
      if (data.status) {
        this.appTypes = data.app_types
        for (var i = this.appTypes.length - 1; i >= 0; i--) {
          this.appTypes[i].apps = []
        }
        this.getApps()
      } else {
        common.alertBox(data.message)
      }
    },
    getApps: function () {
      common.ajax('get', '/user/apps', this.afterGetApps, {}, 'json')
    },
    afterGetApps: function (data) {
      console.log(data)
      for (let i = data.app.length - 1; i >= 0; i--) {
        for (let j = data.app.length - 1; j >= 0; j--) {
          let midd
          if (data.app[i].order > data.app[j].order) {
            midd = data.app[j]
            data.app[j] = data.app[i]
            data.app[i] = midd
          }
        }
      }
      for (let i = data.app.length - 1; i >= 0; i--) {
        let idnum = 'id' + data.app[i].app_type_id
        this[idnum].push(data.app[i])
        this.appTypes[data.app[i].app_type_id - 1].apps.push(data.app[i])
      }
      this.done = true
      common.getCodeList()
    },
    appRouter: function (event) {
      let page = event.currentTarget.getAttribute('router')
      router.go(page)
    }
  },
