import common from '../common.js'
import jgBar from './components/bars/jgBar.vue'
import {changePeopleNumber} from '../vuex/actions.js'
import studentMsg from './components/studentMsg.vue'
export default {
  data () {
    return {
      cid: null,
      bjmc: null,
      xb: JSON.parse(localStorage.getItem('xb')),
      xslb: JSON.parse(localStorage.getItem('xslb')),
      mz: JSON.parse(localStorage.getItem('mz')),
      gb: JSON.parse(localStorage.getItem('gb')),
      pyfs: JSON.parse(localStorage.getItem('pyfs')),
      zzmm: JSON.parse(localStorage.getItem('zzmm')),
      studentList: null,
      addStudentMsg: null,
      activityStudent: null,
      oneStudentMsg: {
        basic: null,
        roll: null,
        family: null
      },
      editStudentMsg: {
        basic: null,
        family: null,
        toll: null
      }
    }
  },
  components: {
    jgBar,
    studentMsg
  },
  methods: {
    doneThing: function (data) {
      console.log(data)
      if (data.status) {
        this.getStudentList()
      }
      common.alertBox(data.message)
    },
    toActivityStudent: function (event) {
      let cid = event.currentTarget.getAttribute('cid')
      this.activityStudent = cid
    },
    getQuery: function () {
      this.bjmc = router._currentRoute.query.bjmc
      this.cid = router._currentRoute.query.cid
      this.getStudentList()
    },
    getStudentList: function () {
      common.ajax('get', '/info/term_class/students/' + this.cid, this.setStudentList)
    },
    setStudentList: function (data) {
      console.log(data)
      if (data.status) {
        this.studentList = data.students
      } else {
        common.alertBox(data.message)
      }
    },
    addStudent: function () {
      let data = {
        xh: this.addStudentMsg
      }
      let jsonData = JSON.stringify(data)
      common.ajax('post', '/info/term_class/students/' + this.cid, this.doneThing, jsonData, 'json')
    },
    deleteStudent: function () {
      common.ajax('delete', '/info/term_class/students/' + this.activityStudent, this.doneThing)
    },
    getStudentMsg: function (event) {
      this.toActivityStudent(event)
      this.changePeopleNumber(event.currentTarget.getAttribute('cid'))
      common.ajax('get', '/info/student/basic/' + this.activityStudent, this.setStudentMsgBasic)
      common.ajax('get', '/info/student/school_roll/' + this.activityStudent, this.setStudentMsgRoll)
      common.ajax('get', '/info/student/family/' + this.activityStudent, this.setStudentMsgFamily)
    },
    setStudentMsgBasic: function (data) {
      console.log(data)
      if (data.status) {
        this.oneStudentMsg.basic = data.basic_info
        this.$broadcast('jgBar-msg', this.oneStudentMsg.basic)
      } else {
        common.alertBox(data.message)
      }
    },
    setStudentMsgRoll: function (data) {
      console.log(data)
      if (data.status) {
        this.oneStudentMsg.roll = data.school_roll.reverse()
      } else {
        common.alertBox(data.message)
      }
    },
    setStudentMsgFamily: function (data) {
      console.log(data)
      if (data.status) {
        this.oneStudentMsg.family = data.family
        this.$broadcast('family-msg', this.oneStudentMsg.family)
      } else {
        common.alertBox(data.message)
      }
    }
  },
  vuex: {
    actions: {
      changePeopleNumber
    }
  },
  ready () {
    this.getQuery()
  }
}
