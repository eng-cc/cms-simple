import common from '../common.js'
export default {
  data () {
    return {
      collegeList: null,
      majorList: null,
      classList: null,
      activityCollege: null,
      activityMajor: null,
      activityClass: null,
      activityMajorName: null,
      oneMajorMsg: null,
      xueqi: JSON.parse(localStorage.getItem('xueqi')),
      xiaoqu: JSON.parse(localStorage.getItem('xiaoqu')),
      addClassMsg: {
        xqdm: null,
        xy_code: null,
        zy_code: null,
        nj: null,
        bh: null,
        bjmc: null,
        xq_code: null,
        qqq: null,
        sfsyb: null
      },
      oneClassMsg: null,
      editClassMsg: {
        xqdm: null,
        xy_code: null,
        zy_code: null,
        nj: null,
        bh: null,
        bjmc: null,
        xq_code: null,
        qqq: null,
        sfsyb: null
      }
    }
  },
  methods: {
    doneThing: function (data) {
      console.log(data)
      if (data.status) {
        this.getClassList()
        common.alertBox(data.message)
      } else {
        common.alertBox(data.message)
      }
    },
    activityOneMajor: function (event) {
      let cid = event.currentTarget.getAttribute('cid')
      this.activityMajor = cid
    },
    getCollegeList: function () {
      common.ajax('get', '/info/colleges', this.setCollegeList, {}, 'json')
    },
    setCollegeList: function (data) {
      console.log(data)
      if (data.status) {
        this.collegeList = data.colleges
        this.activityCollege = this.collegeList[0].yxsh
        this.getMajorList()
      } else {
        common.alertBox(data.message)
      }
    },
    getMajorList: function (a) {
      if (a === 2) {
        this.activityCollege = this.addClassMsg.xy_code
      };
      let data = {
        xy_code: this.activityCollege
      }
      let jsonData = JSON.stringify(data)
      common.ajax('post', '/info/major_list', this.setMajorList, jsonData, 'json')
    },
    setMajorList: function (data) {
      console.log(data)
      if (data.status) {
        this.majorList = data.majors
        this.activityMajor = this.majorList[0].zyh
        this.getClassList()
      } else {
        common.alertBox(data.message)
      }
    },
    getClassList: function () {
      let data = {
        xy_code: this.activityCollege,
        zy_code: this.activityMajor
      }
      let jsonData = JSON.stringify(data)
      common.ajax('post', '/info/class_list', this.setClassList, jsonData, 'json')
    },
    setClassList: function (data) {
      console.log(data)
      if (data) {
        this.classList = data.classes
        for (let i = 0; i < this.majorList.length; i++) {
          if (this.majorList[i].zyh === this.activityMajor) {
            this.activityMajorName = this.majorList[i].zymc
            break
          }
        }
      } else {
        common.alertBox(data.message)
      };
    },
    toActivityClass: function (event) {
      let cid = event.currentTarget.getAttribute('cid')
      this.activityClass = cid
    },
    addClass: function () {
      let jsonData = JSON.stringify(this.addClassMsg)
      this.activityMajor = this.addClassMsg.xy_code
      common.ajax('post', '/info/classes', this.doneThing, jsonData, 'json')
    },
    getOneClass: function (event) {
      this.toActivityClass(event)
      common.ajax('get', '/info/class/' + this.activityClass, this.setOneClassMsg, {}, 'json')
    },
    setOneClassMsg: function (data) {
      console.log(data)
      if (data.status) {
        this.oneClassMsg = data.class
      } else {
        common.alertBox(data.message)
      };
    },
    editClass: function () {
      let jsonData = JSON.stringify(this.editClassMsg)
      common.ajax('put', '/info/class/' + this.activityClass, this.doneThing, jsonData, 'json')
    },
    deleteClass: function () {
      common.ajax('delete', '/info/class/' + this.activityClass, this.doneThing, {}, 'json')
    }
  },
  ready () {
    this.getCollegeList()
  }
