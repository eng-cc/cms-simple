import common from '../common.js'
export default {
  data () {
    return {
      appTypes: Array,
      groupList: null,
      activeCid: null,
      activeName: null,
      activityLevel: null,
      activityDescription: null,
      addGroup: {
        name: null,
        description: null
      },
      editGroup: {
        name: null,
        description: null,
        gldj: null
      },
      oneAppMsg: null,
      geted: false
    }
  },
  methods: {
    doneThing: function (data) {
      console.log(data)
      if (data.status) {
        this.getGroupList()
      };
      common.alertBox(data.message)
    },
    getGroupList: function () {
      this.geted = false
      common.ajax('get', '/admin/roles', this.useGroupList, {}, 'json')
    },
    useGroupList: function (data) {
      if (data.status) {
        console.log(data)
        this.groupList = data.roles
        this.geted = true
      } else {
        common.alertBox(data.message)
      }
    },
    changeActiveCid: function () {
      let cid = event.currentTarget.getAttribute('cid')
      let cname = event.currentTarget.getAttribute('cname')
      let cdis = event.currentTarget.getAttribute('cdis')
      // let type = event.currentTarget.getAttribute('type')
      this.activeCid = cid
      this.activeName = cname
      this.activityDescription = cdis
      // this.activityLevel = type.length > 0 ? type : null
    },
    editOneGroup: function () {
      let jsonData = JSON.stringify(this.editGroup)
      common.ajax('put', '/admin/role/' + this.activeCid, this.doneThing, jsonData, 'json')
    },
    deleteOneGroup: function () {
      let url = '/admin/role/' + this.activeCid
      common.ajax('delete', url, this.doneThing, {}, 'json')
    },
    addOneGroup: function () {
      var jsonData = JSON.stringify(this.addGroup)
      common.ajax('post', '/admin/roles', this.doneThing, jsonData, 'json')
    }
  },
  ready () {
    this.getGroupList()
  }
}
