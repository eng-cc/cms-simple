import common from '../common.js'
export default {
  data () {
    return {
      isteacher: null,
      isstudent: null,
      basic: null,
      schoolRoll: [],
      family: null,
      complete: 0,
      picUrl: localStorage.getItem('picurl')
    }
  },
  methods: {
    getMsgs: function (url, cb) {
      common.ajax('get', url, cb, {}, 'json')
    },
    useBasic: function (data) {
      console.log(data)
      if (data.status) {
        if (data.is_teacher) {
          this.isteacher = true
          this.complete = 3
        } else {
          this.isstudent = true
        }
        this.basic = data.base_info
        this.complete += 1
        if (!data.is_teacher) {
          this.getMsgs('/user/profile/school_roll', this.useSchoolRoll)
          this.getMsgs('/user/profile/family', this.useFamily)
        }
      } else {
        common.alertBox(data.message)
      }
      console.log(this.picUrl + '+++++++++')
    },
    useFamily: function (data) {
      console.log(data)
      if (data.status) {
        // console.log(this.family);
        this.complete += 1
        this.family = data.family_info[0]
        // console.log(this.family[0].jtzz);
      } else {
        common.alertBox(data.message)
      }
    },
    useSchoolRoll: function (data) {
      console.log(data)
      if (data.status) {
        for (let i = data.school_roll.length - 1; i >= 0; i--) {
          this.schoolRoll.push(data.school_roll[i])
        }
        this.complete += 1
      } else {
        common.alertBox(data.message)
      }
    }
  },
  ready () {
    this.getMsgs('/user/profile/basic', this.useBasic)
  }
}
