import common from '../common.js'
export default {
  data () {
    return {
      beginDay: null,
      endDay: null,
      recordList: null,
      detail: null,
      beginTime: null,
      endTime: null,
      leaveType: null,
      geted: false,
      pages: null,
      prevNum: null,
      nextNum: null,
      activityMember: null,
      activityLeaveStatus: null,
      activityPage: null
    }
  },
  methods: {
    doneThing: function (data) {
      console.log(data)
      this.getPage(1)
      common.alertBox(data.message)
    },
    getPage (num) {
      console.log('567')
      this.activityPage = num
      console.log(num)
      common.ajax('get', '/leave/approval/' + num, this.selectedDay)
    },
    selectedDay: function (data) {
      console.log(data)
      if (data.status) {
        this.geted = true
        this.recordList = data.result
        this.prevNum = data.prev
        this.nextNum = data.next
        this.pages = data.total
        this.activityLeaveStatus = 2
      } else {
        common.alertBox(data.message)
      }
    },
    getAdmin (event) {
      this.activityMember = event.currentTarget.getAttribute('cid')
      console.log(this.activityMember)
    },
    detailMsg: function (event) {
      let bh = event.currentTarget.getAttribute('bh')
      for (let i = 0; i < this.recordList.length; i++) {
        if (bh === this.recordList[i].bh) {
          this.detail = this.recordList[i].qjnr
          this.beginTime = this.recordList[i].kssj
          this.endTime = this.recordList[i].jssj
          this.leaveType = this.recordList[i].qjlb
          console.log(this.detail)
        }
      }
    },
    PostLeaveStatus () {
      console.log(this.activityMember)
      let data = {
        sqzt_code: this.activityLeaveStatus
      }
      console.log(data)
      let jsonData = JSON.stringify(data)
      common.ajax('put', '/leave/approval/manage/' + this.activityMember, this.doneThing, jsonData, 'json')
    }
  },
  ready () {
    common.datepickerInit()
    this.getPage(1)
  }
}
