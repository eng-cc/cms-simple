import common from '../../common.js'
import {getLocation} from '../../vuex/actions.js'
export default {
  data () {
    return {
      position: null
    }
  },
  methods: {
    mapInit: function () {
      let that = this
      let mapObj = new AMap.Map('iCenter')
      mapObj.plugin('AMap.Geolocation', function () {
        let geolocation = new AMap.Geolocation({
          enableHighAccuracy: true, // 是否使用高精度定位，默认:true
          timeout: 10000, // 超过10秒后停止定位，默认：无穷大
          maximumAge: 100 // 定位结果缓存0毫秒，默认：0
        })
        mapObj.addControl(geolocation)
        geolocation.getCurrentPosition()
        AMap.event.addListener(geolocation, 'complete', that.onComplete) // 返回定位信息
        AMap.event.addListener(geolocation, 'error', that.onError) // 返回定位出错信息
      })
      console.log('init')
    },
    onComplete: function (data) {
      console.log(data)
      this.position = data.position
      this.getFullLocation()
    },
    onError: function (data) {
      console.log(data)
      common.alertBox('error')
    },
    // getFullLocation: function() {
      //     let url = ' http://restapi.amap.com/v3/geocode/regeo?output=json&' +
      //      'key=789580d1d7f8c2d97574ae1d84b0104d&location=' + this.position.lng +
      //      ',' + this.position.lat;
      //      common.ajax('get',url,this.setFullLocation)
      // },
    getFullLocation: function () {
      let that = this
      // let map = new AMap.Map('iCenter')
      let lnglatXY = [this.position.lng, this.position.lat] // 点坐标
        // 逆地理编码
      let geocoder = new AMap.Geocoder({
        radius: 1000,
        extensions: 'all'
      })
      geocoder.getAddress(lnglatXY, function (status, result) {
        if (status === 'complete' && result.info === 'OK') {
          that.setFullLocation(result)
        }
      })
    },
    setFullLocation: function (data) {
      console.log(data.regeocode)
      console.log(data.regeocode.formattedAddress)
        // this.location = data.regeocode.formattedAddress
      this.getLocation(data.regeocode.formattedAddress)
    }
  },
  vuex: {
    actions: {
      getLocation
    }
  },
  ready () {
    this.mapInit()
  }
}
// </script>
