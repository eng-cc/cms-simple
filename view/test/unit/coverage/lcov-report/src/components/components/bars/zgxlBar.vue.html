//       <select name="searchtype" class="form-control" v-model="activitySecCode" :value="parentMsg.sec">

import common from '../../../common.js'
export default {
  data () {
    return {
      secCode: null,
      activityFirCode: null,
      firCode: null,
      activitySecCode: null,
      parcode: null,
      parentMsg: {
        fir: null,
        sec: null
      },
      isSet: true
    }
  },
  methods: {
    getFirCode: function () {
      let data = {
        level: 1,
        code: null
      }
      let jsonData = JSON.stringify(data)
      common.ajax('post', '/info/code_zgxl', this.setFirCode, jsonData, 'json')
    },
    setFirCode: function (data) {
      console.log(data)
      if (data.status) {
        this.firCode = data.code
        this.activityFirCode = this.parentMsg.fir == null ? this.firCode[0].dm : this.parentMsg.fir
        this.getSecCode()
      } else {
        common.alertBox(data.message)
      };
    },
    getSecCode: function () {
      let data = {
        level: 2,
        code: this.activityFirCode
      }
      let jsonData = JSON.stringify(data)
      common.ajax('post', '/info/code_zgxl', this.setSecCode, jsonData, 'json')
    },
    setSecCode: function (data) {
      console.log(data)
      if (data.status) {
        this.secCode = data.code
        if (this.isSet) {
          this.parentMsg.sec = this.parcode
          this.activitySecCode = this.parcode
        } else {
          this.parentMsg.sec = this.secCode[0].dm
          this.activitySecCode = this.secCode[0].dm
        }
        this.isSet = false
      } else {
        common.alertBox(data.message)
      };
    }
  },
  watch: {
    zgxlCode () {
      this.getFirCode()
      this.isSet = true
      if (this.zgxlCode) {
        this.parcode = this.zgxlCode
        if (this.parcode === '02' || this.parcode === '03') {
          this.parentMsg.fir = '01'
        } else if (this.parcode === '06' || this.parcode === '07') {
          this.parentMsg.fir = '05'
        } else if (this.parcode === '10' || this.parcode === '11' || this.parcode === '19') {
          this.parentMsg.fir = '10'
        } else if (this.parcode === '20' || this.parcode === '21' || this.parcode === '28' || this.parcode === '29') {
          this.parentMsg.fir = '20'
        };
        this.activityFirCode = this.parentMsg.fir
      }
    }
  },
  vuex: {
    getters: {
      zgxlCode: ({selectBars}) => selectBars.zgxlBar
    }
  },
  ready () {
 
  }
}
