import common from '../../common.js'
import jgBar from './bars/jgBar.vue'
import zcjbBar from './bars/zcjbBar.vue'
import zgxlBar from './bars/zgxlBar.vue'
import {changeJgBar, changeZcjbBar, changeZgxlBar} from '../../vuex/actions.js'
export default {
  data () {
    return {
      clear: false,
      oneTeacherMsg: null,
      codeList: {
        gb: JSON.parse(localStorage.getItem('gb')),
        xw: JSON.parse(localStorage.getItem('xw')),
        xzjb: JSON.parse(localStorage.getItem('xzjb')),
        zzmm: JSON.parse(localStorage.getItem('zzmm')),
        grsf: JSON.parse(localStorage.getItem('grsf')),
        mz: JSON.parse(localStorage.getItem('mz')),
        xb: JSON.parse(localStorage.getItem('xb'))
      },
      departmentList: null,
      max_role_level: localStorage.getItem('max_role_level'),
      editTeacherMsg: {
        xb_code: null,
        mz_code: null,
        csny: null,
        sfzh: null,
        jg_code: null,
        gzny: null,
        szdw_code: null,
        zw: null,
        xzjb_code: null,
        zcjb_code: null,
        zgxl_code: null,
        xw_code: null,
        grsf_code: null,
        zzmm_code: null
      }
    }
  },
  components: {
    jgBar,
    zgxlBar,
    zcjbBar
  },
  methods: {
    doneThing: function (data) {
      console.log(data)
      if (data.status) {
        // this.getApplyList()
      };
      common.alertBox(data.message)
    },
    getDepartmentList: function () {
      common.ajax('get', '/info/departments', this.useDepartmentList, {}, 'json')
    },
    useDepartmentList: function (data) {
      if (data.status) {
        console.log(data)
        this.departmentList = data.departments
      } else {
        common.alertBox(data.message)
      }
    },
    getOneTeacher: function () {
      this.oneTeacherMsg = null
      common.ajax('get', '/info/teacher/' + this.teacherNumber, this.setOneTeacher)
    },
    setOneTeacher: function (data) {
      if (data.status) {
        console.log(data)
        this.oneTeacherMsg = data.teacher
        this.oneTeacherMsg.show = true
        this.changeJgBar(data.teacher.jg.dm)
        this.changeZcjbBar(data.teacher.zcjb.dm)
        this.changeZgxlBar(data.teacher.zgxl.dm)
        console.log(data.teacher.zcjb.dm)
        this.$broadcast('getde-msg', this.oneTeacherMsg)
        this.$broadcast('jgBar-msg', this.oneTeacherMsg)
      } else {
        common.alertBox(data.message)
      }
    },
    toEditTeacherMsg: function () {
      console.log(this.$refs)
      let jgCode = this.$refs.jgbar.activityThrCode
      let zgxlCode = this.$refs.zgxlbar.activitySecCode
      let zcjbCode = this.$refs.zcjbbar.activitySecCode
      this.editTeacherMsg.jg_code = jgCode
      this.editTeacherMsg.zgxl_code = zgxlCode
      this.editTeacherMsg.zcjb_code = zcjbCode
      let jsonData = JSON.stringify(this.editTeacherMsg)
      common.ajax('put', '/info/teacher/' + this.teacherNumber, this.doneThing, jsonData, 'json')
    },
    toEditTeacherBasicMsg: function () {
      if (this.editTeacherBasicMsg.jhkh && this.editTeacherBasicMsg.jhkh.length === 19) {
        if (this.editTeacherBasicMsg.sjh && this.editTeacherBasicMsg.sjh.length === 11) {
          let jsonData = JSON.stringify(this.editTeacherBasicMsg)
          common.ajax('put', '/info/teacher/contact/' + this.teacherNumber, this.doneThing, jsonData, 'json')
        } else {
          common.alertBox('手机号不正确')
        }
      } else {
        common.alertBox('建行卡号不正确')
      }
    }
  },
  watch: {
    teacherNumber: function () {
      this.getOneTeacher()
    }
  },
  vuex: {
    getters: {
      teacherNumber: ({ peopleMsg }) => peopleMsg.peopleNumber,
      jgCodeVuex: ({ selectBars }) => selectBars.jgBar,
      zgxlCodeVuex: ({ selectBars }) => selectBars.zgxlBar,
      zcjbCodeVuex: ({ selectBars }) => selectBars.zcjbBar
    },
    actions: {
      changeZgxlBar,
      changeJgBar,
      changeZcjbBar
    }
  },
  ready () {
    this.getDepartmentList()
  }
}
