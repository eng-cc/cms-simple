import common from '../common.js'
export default {
  data () {
    return {
      recordList: null,
      leaveType: null,
      activityMember: null,
      geted: false,
      askleave: {
        activityLeaveType: null,
        qjnr: null,
        kssj: null,
        jssj: null
      },
      detail: null
    }
  },
  methods: {
    doneThing: function (data) {
      console.log(data)
     // console.log('345')
      this.getaskLeave()
      this.askleave.qjnr = null
      this.askleave.kssj = null
      this.askleave.jssj = null
      common.alertBox(data.message)
    },
    getDate () {
      let date = new Date()
      let month
      let day
      if ((date.getMonth() + 1) < 10) {
        month = '0' + (date.getMonth() + 1)
      }
      if (date.getDate() < 10) {
        day = '0' + date.getDate()
      } else {
        day = date.getDate()
      }
      this.time = date.getFullYear() + '-' + month + '-' + day
      console.log(this.time)
      console.log(123)
    },
    getLeaveType () {
      common.ajax('get', '/leave/askleave/type', this.setLeaveType)
    },
    setLeaveType: function (data) {
      console.log(data)
      if (data.status) {
        this.leaveType = data.result
        this.askleave.activityLeaveType = this.leaveType[0].dm
        console.log(this.askleave.activityLeaveType)
      } else {
        common.alertBox(data.message)
      }
    },
    getAdmin (event) {
      this.activityMember = event.currentTarget.getAttribute('cid')
      console.log(this.activityMember)
    },
    // getStatus () {
    //   common.ajax('get', '/leave/askleave/status', this.doneThing)
    // },
    getaskLeave () {
      common.ajax('get', '/leave/askleave', this.setaskLeave)
    },
    setaskLeave: function (data) {
      console.log(data)
      if (data.status) {
        this.recordList = data.result
        this.geted = true
      } else {
        common.alertBox(data.message)
      }
    },
    detailMsg: function (event) {
      let bh = event.currentTarget.getAttribute('bh')
      for (let i = 0; i < this.recordList.length; i++) {
        if (bh === this.recordList[i].bh) {
          this.detail = this.recordList[i].qjnr
          console.log(this.detail)
        }
      }
    },
    PostAskLeave () {
      let data = {
        qjlb_code: this.askleave.activityLeaveType,
        qjnr: this.askleave.qjnr,
        kssj: this.askleave.kssj,
        jssj: this.askleave.jssj
      }
      console.log(data)
      let jsonData = JSON.stringify(data)
      common.ajax('post', '/leave/askleave', this.doneThing, jsonData, 'json')
    },
    deleteLeave () {
      console.log(this.activityMember)
      common.ajax('delete', '/leave/upleave/' + this.activityMember, this.doneThing, {}, 'json')
    },
    returnLeave () {
      console.log(this.activityMember)
      common.ajax('put', '/leave/upleave/' + this.activityMember, this.doneThing, {}, 'json')
    }
  },
  ready () {
    common.datepickerInit()
    this.getDate()
    this.getLeaveType()
    this.getaskLeave()
  }
}
