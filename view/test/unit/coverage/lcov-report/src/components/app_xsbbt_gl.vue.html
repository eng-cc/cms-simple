import common from '../common.js'
export default {
  data () {
    return {
      classList: null,
      activityTerm: null,
      xueqi: null,
      activityTermClass: null,
      applyList: null,
      isPass: 1,
      geted: false
    }
  },
  methods: {
    doneThing: function (data) {
      console.log(data)
      if (data.status) {
        this.getApplyList()
      };
      common.alertBox(data.message)
    },
    toActivityCid (event) {
      this.activityCid = event.currentTarget.getAttribute('cid')
    },
    getTermList () {
      common.ajax('get', '/supp/spark/term', this.setTermList)
    },
    setTermList (data) {
      console.log(data)
      if (data.status) {
        this.xueqi = data.terms
      } else {
        common.alertBox(data.message)
      }
      if (this.xueqi.length > 0) {
        this.activityTerm = this.xueqi[0].xqdm
        this.getClassList()
      } else {
        this.geted = true
        common.alertBox('所带班级没有申请')
      }
    },
    getClassList () {
      this.applyList = null
      common.ajax('get', '/info/class_ins_list/' + this.activityTerm, this.setClassList)
    },
    setClassList (data) {
      console.log(data)
      if (data.status) {
        this.classList = data.classes
        if (this.classList.length > 0) {
          this.activityTermClass = this.classList[0].id
          this.getApplyList()
        }
      } else {
        common.alertBox(data.message)
      }
    },
    getApplyList () {
      this.geted = false
      common.ajax('get', '/supp/spark/manages/' + this.activityTermClass, this.setApplyList)
    },
    setApplyList (data) {
      console.log(data)
      if (data.status) {
        this.geted = true
        this.applyList = data.students
      }
    },
    changeApply () {
      if (+this.isPass === 1) {
        common.ajax('post', '/supp/spark/manage/' + this.activityCid, this.doneThing)
      } else {
        common.ajax('delete', '/supp/spark/manage/' + this.activityCid, this.doneThing, {}, 'json')
      }
      this.geted = false
    }
  },
  ready () {
    this.getTermList()
  }
}
