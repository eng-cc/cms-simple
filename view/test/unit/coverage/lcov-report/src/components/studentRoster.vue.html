import common from '../common.js'
import studentMsg from './components/studentMsg.vue'
import familyBar from './components/bars/familyBar.vue'
import {changePeopleNumber} from '../vuex/actions.js'
// console.log(common)
export default {
  data (data) {
    return {
      classList: null,
      activityTerm: null,
      xueqi: JSON.parse(localStorage.getItem('xueqi')),
      studentList: null,
      activityClass: null,
      geted: false
    }
  },
  components: {
    studentMsg: studentMsg,
    familyBar: familyBar
  },
  methods: {
    setDefauleSelect: function () {
      this.activityTerm = this.xueqi[0].dm
      this.getClassList()
    },
    getClassList: function () {
      this.studentList = null
      common.ajax('get', '/info/class_ins_list/' + this.activityTerm, this.setClassList)
    },
    setClassList: function (data) {
      console.log(data)
      if (data.status) {
        this.classList = data.classes
        if (this.classList.length > 0) {
          this.activityClass = this.classList[0].id
          this.getStudentList()
        }
      } else {
        common.alertBox(data.message)
      }
    },
    getStudentList: function (event) {
      this.geted = false
      common.ajax('get', '/info/students_ins_list/' + this.activityClass, this.setStudentList)
    },
    setStudentList: function (data) {
      console.log(data)
      if (data.status) {
        this.geted = true
        this.studentList = data.students
      } else {
        common.alertBox(data.message)
      }
    },
    activityOneStudent: function (event) {
      this.activityStudent = event.currentTarget.getAttribute('cid')
    },
    getStudentMsg: function (event) {
      this.activityOneStudent(event)
      this.changePeopleNumber(event.currentTarget.getAttribute('cid'))
    }
  },
  vuex: {
    actions: {
      changePeopleNumber
    }
  },
  ready () {
    this.setDefauleSelect()
  }
}
