import common from '../common.js'
export default {
  data () {
    return {
      picurl: localStorage.getItem('picurl'),
      type: null,
      name: null,
      url: null,
      file: null,
      uploading: false,
      isselect: false
    }
  },
  methods: {
    getImage (event) {
      // docs
      // https://www.awesomes.cn/repo/fengyuanchen/cropper
      // http://www.htmleaf.com/jQuery/Image-Effects/201504211716.html
      this.url = null
      let el = event.currentTarget
      let file = el.files[0]
      if (file.type === 'image/jpeg') {
        this.name = file.name
        this.type = file.type
        this.file = true
        let url = window.URL || window.webkitURL
        let originPhotoUrl = url.createObjectURL(file)
        $('#cropperImg').cropper({
          aspectRatio: 3 / 4,
          autoCropArea: 1,
          rotatable: false,
          minContainerWidth: 120,
          minContainerHeight: 160
        }).cropper('replace', originPhotoUrl)
      } else {
        common.alertBox('只允许上传jpg格式照片')
      }
    },
    updateImg () {
      if (this.file) {
        this.uploading = true
        let bs = $('#cropperImg').cropper('getCroppedCanvas', {
          width: 330,
          height: 440
        }).toDataURL('image/jpeg', 1.0)
        console.log(bs.length)
        if (bs.length > 800000) {
          common.alertBox('图片过大，请压缩后上传')
        } else {
          this.url = bs
        }
        let data = {
          encoded_pic: this.url,
          xh: localStorage.getItem('username')
        }
        let jsonData = JSON.stringify(data)
        common.ajax('post', '/info/students/change_pic', this.afterUpload, jsonData, 'json')
      }
    },
    afterUpload (data) {
      console.log(data)
      if (data.status) {
        common.alertBox('上传成功')
      } else {
        // common.alertBox(data.message)
      }
      this.uploading = false
    },
    clickInput () {
      $('#inputbtn').click()
    }
  },
  ready () {}
}
