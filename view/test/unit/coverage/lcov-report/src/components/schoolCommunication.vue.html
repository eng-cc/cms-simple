import common from '../common.js'
export default {
  data () {
    return {
      workPlace: null,
      activityWorkPlace: localStorage.getItem('userDepartmentId'),
      teacherCommunication: null,
      actvityPic: null,
      geted: false
    }
  },
  methods: {
    doneThing: function (data) {
      console.log(data)
      if (data.status) {
        this.getworkPlace()
      };
      common.alertBox(data.message)
    },
    getworkPlace: function () {
      common.ajax('get', '/info/contacts/colleges', this.setworkPlace)
    },
    setworkPlace: function (data) {
      console.log(data)
      if (data.status) {
        this.workPlace = data.contactsc
        // console.log(this.activityWorkPlace)
        console.log(this.activityWorkPlace)
        this.geTeacherCommunication()
        // console.log(this.workPlace)
      } else {
        common.alertBox(data.message)
      }
    },
    geTeacherCommunication: function (data) {
      this.geted = false
      common.ajax('get', '/info/contacts/college/' + this.activityWorkPlace, this.setTeacherCommunication)
    },
    setTeacherCommunication: function (data) {
      console.log(data)
      if (data.status) {
        // 增加备注里面内容为所带班级，并进行字符串分割，方便渲染
        for (let i = data.contacts.length - 1; i >= 0; i--) {
          data.contacts[i].bzArr = data.contacts[i].bz.split(',')
          data.contacts[i].bzArr.pop()
        }
        this.teacherCommunication = data.contacts
        console.log(this.teacherCommunication)
        this.teacherCommunication = data.contacts.reverse()
        this.geted = true
      } else {
        common.alertBox(data.message)
      }
    },
    bigPic: function (event) {
      let zgh = event.currentTarget.getAttribute('zgh')
      for (let i = 0; i < this.teacherCommunication.length; i++) {
        if (zgh === this.teacherCommunication[i].zgh) {
          this.actvityPic = this.teacherCommunication[i].pic_url
          console.log(this.actvityPic)
        }
      }
    }
  },
  ready () {
    this.getworkPlace()
  }
}
