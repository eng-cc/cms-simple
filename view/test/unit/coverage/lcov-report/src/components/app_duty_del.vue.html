      workPlace: null,

      dayMsg: {
        activityDepartment: null,
        activityPeople: null,
        activityTime: null
      },
      departmentMsg: {
        activityDepartment: null,
        activityTime: null,
        activityTime2: null
      },
      timeMsg: {
        activityDepartment: null,
        activityPeople: null,
        activityTime: null,
        activityTime2: null
      }
    }
  },
  methods: {
    doneThing: function (data) {
      console.log(data)
      if (data.status) {
        // this.getApplyList()
      }
      common.alertBox(data.message)
    },
    getDate () {
      let date = new Date()
      this.activityTime = this.activityTime2 = date.getFullYear() + '-' + (date.getMonth() + 1) + '-' + date.getDate()
    },
    getworkPlace: function () {
      common.ajax('get', '/duty/department', this.setworkPlace)
    },
    setworkPlace: function (data) {
      console.log(data)
      if (data.status) {
        this.workPlace = data.contactsc
      } else {
        common.alertBox(data.message)
      }
    },
    getDepartmentPeoples (obj) {
      common.ajax('get', '/duty/teachers/' + obj.activityDepartment, this.setDepartmentPeoples)
    },
    setDepartmentPeoples (data) {
      console.log(data)
      if (data.status) {
        this.departmentPeoples = data.contacts
      } else {
        common.alertBox(data.message)
      }
    },
    delByDay () {
      let data = {
        dw_code: this.dayMsg.activityDepartment,
        zgh_code: this.dayMsg.activityPeople,
        zbrq: this.dayMsg.activityTime
      }
      let jsonData = JSON.stringify(data)
      common.ajax('put', '/duty/arrange/day', this.doneThing, jsonData, 'json')
    },
    delByTime () {
      let data = {
        dw_code: this.timeMsg.activityDepartment,
        zgh_code: this.timeMsg.activityPeople,
        bzbrq: this.timeMsg.activityTime,
        ezbrq: this.timeMsg.activityTime2
      }
      let jsonData = JSON.stringify(data)
      common.ajax('put', '/duty/arrange/time', this.doneThing, jsonData, 'json')
    },
    delByAll () {
      let data = {
        yxsh: this.departmentMsg.activityDepartment,
        begin: this.departmentMsg.activityTime,
        end: this.departmentMsg.activityTime2
      }
      let jsonData = JSON.stringify(data)
      common.ajax('put', '/duty/arrange/all', this.doneDelThing, jsonData, 'json')
    },
    doneDelThing: function (data) {
      console.log(data)
      common.alertBox(data.message, data.status)
    }
