import common from '../common.js'
export default {
  data () {
    return {
      studentNum: null,
      activityCollege: 50,
      showLoading: false,
      studentSelfList: null,
      collegeList: null,
      activityTerm: null,
      xueqi: JSON.parse(localStorage.getItem('xueqi')),
      nianji: JSON.parse(localStorage.getItem('nianji')),
      termClassList: null,
      activityNianji: null,
      activityXueqi: null,
      activityTermClass: null,
      geted: true
    }
  },
  methods: {
    doneThing: function (data) {
      console.log(data)
      if (data.status) {
        this.getStudentSelfList()
      };
      common.alertBox(data.message)
    },
    getCollegeList: function () {
      if (localStorage.getItem('college')) {
        console.log(localStorage.getItem('college'))
        this.collegeList = JSON.parse(localStorage.getItem('college'))
      } else {
        common.ajax('get', '/info/colleges', this.setCollegeList, {}, 'json')
      }
    },
    setCollegeList: function (data) {
      console.log(data)
      if (data.status) {
        this.collegeList = data.colleges
        localStorage.setItem('college', JSON.stringify(data.colleges))
      } else {
        common.alertBox(data.message)
      }
    },
    getStudentSelfList: function () {
      this.geted = false
      common.ajax('get', '/info/students/school_roll/' + this.studentNum, this.setStudentSelfList)
    },
    setStudentSelfList: function (data) {
      console.log(data)
      if (data.status) {
        data.sudents = data.students.reverse()
        console.log(data.sudents)
        this.studentSelfList = data.students
        console.log(this.studentSelfList)
        this.geted = true
      } else {
        common.alertBox(data.message)
      }
    },
    getOneTerm: function (event) {
      let num = event.currentTarget.getAttribute('num')
      this.activityTerm = this.studentSelfList[num]
    },
    geTermClassList: function () {
      let data = {
        xqdm: this.activityXueqi,
        xy_code: this.activityCollege,
        nj: this.activityNianji,
        level: 2
      }
      let jsonData = JSON.stringify(data)
      common.ajax('post', '/info/students/school_roll/term_classes', this.setTermClassList, jsonData, 'json')
    },
    setTermClassList: function (data) {
      console.log(data)
      if (data.status) {
        this.termClassList = data.term_classes
      } else {
        common.alertBox(data.message)
      }
    },
    editSchoolRoll: function () {
      let data = {
        old_xqbjid: this.activityTerm.xqbjid,
        xqbjid: this.activityTermClass
      }
      let jsonData = JSON.stringify(data)
      common.ajax('put', '/info/students/school_roll/' + this.studentNum, this.doneThing, jsonData, 'json')
    }
  },
  ready () {
    this.getCollegeList()
  }
}
