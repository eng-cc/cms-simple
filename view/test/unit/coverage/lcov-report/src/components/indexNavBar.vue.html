      // let userName = localStorage.getItem('username') ? localStorage.getItem('username') : null
      common.ajax('get', '/user/profile/index', this.setUserMsg, {}, 'json')
    },
    setUserMsg: function (data) {
      console.log(data)
      if (data.status) {
        this.userMsg = data
        localStorage.setItem('picurl', data.head_pic_url)
        localStorage.setItem('userDepartantMc', data.department)
        localStorage.setItem('max_role_level', data.max_role_level)
        localStorage.setItem('userDepartmentId', data.department_id)
      } else {
        common.alertBox(data.message)
      };
    },
    getCurrentTerm: function () {
      common.ajax('get', '/main/current_term', this.setCurrentTerm)
    },
    setCurrentTerm: function (data) {
      // console.log(data.current_term)
      if (data.status) {
        localStorage.setItem('currentTerm', data.current_term)
      }
    }
  },
  ready () {

    this.getUserMsg()
