import common from '../common.js'
export default {
  data () {
    return {
      textList: null,
      allQuestion: null,
      activityId: null,
      activityText: null,
      canTest: false,
      canSubmit: null,
      timeNum: null,
      timeout: null,
      timeType: null,
      answer: null,
      score: null,
      geted: false,
      geted2: false
    }
  },
  methods: {
    doneThing: function (data) {
      console.log(data)
      common.alertBox(data.message)
      if (data.status) {
        this.getTextList()
      }
    },
    clock (time, type) {
      clearInterval(this.timeout)
      this.timeNum = time * 60
      this.timeout = setInterval(() => {
        this.timeNum--
        if (this.timeNum <= 0) {
          clearInterval(this.timeout)
          if (type === 'learn') {
            this.finishOneLearn()
            this.activityText.isfinish = true
          } else if (type === 'test') {
            this.canSubmit = false
          }
        }
      }, 1000)
    },
    getTextList () {
      common.ajax('get', '/freshman/learning/list', this.setTextList)
    },
    setTextList (data) {
      console.log(data)
      if (data.status) {
        this.textList = data.data
        this.geted = true
        for (let i = 0; i < data.data.length; i++) {
          for (let j = 0; j < data.data[i].sections.length; j++) {
            this.canTest = true
            if (!data.data[i].sections[j].isfinish) {
              this.canTest = false
              break
            }
          }
        }
      } else {
        common.alertBox(data.message)
      }
    },
    getOneTxext (event) {
      this.geted2 = false
      this.activityText = null
      let id = event.currentTarget.getAttribute('cid')
      this.activityId = id
      common.ajax('get', '/freshman/learning/' + id, this.setOneText)
    },
    setOneText (data) {
      console.log(data)
      if (data.status) {
        this.activityText = data
        this.geted2 = true
        if (data.isfinish) {
          this.clock(data.time, 'learn')
        }
      } else {
        common.alertBox(data.message)
      }
    },
    finishOneLearn () {
      common.ajax('put', '/freshman/learning/' + this.activityId, this.doneThing)
    },
    getAllQuestion () {
      document.oncontextmenu = new Function('event.returnValue=false;') // eslint-disable-line
      document.onselectstart = new Function('event.returnValue=false;') // eslint-disable-line
      this.allQuestion = null
      common.ajax('get', '/freshman/exam', this.setAllQuestion)
    },
    setAllQuestion (data) {
      console.log(data)
      if (data.status) {
        this.allQuestion = data.data
        this.testId = data.id
        this.testTime = data.time
        this.canSubmit = true
        this.clock(120, 'test')
        this.answer = {}
      } else {
        common.alertBox(data.message)
      }
    },
    pushAnswer (event, answer) {
      let id = event.currentTarget.getAttribute('name')
      console.log(id, answer)
      this.answer[id] = answer
    },
    submitAnswer () {
      document.oncontextmenu = new Function('event.returnValue=true;') // eslint-disable-line
      document.onselectstart = new Function('event.returnValue=true;') // eslint-disable-line
      if (this.canSubmit) {
        let jsonData = JSON.stringify({
          id: this.testId,
          answer: this.answer
        })
        common.ajax('post', '/freshman/exam', this.doneThing, jsonData, 'json')
      }
    },
    getScore () {
      this.score = null
      common.ajax('get', '/freshman/score', this.setScore)
    },
    setScore (data) {
      console.log(data)
      if (data) {
        this.score = data.data
      } else {
        common.alertBox(data.message)
      }
    }
  },
  ready () {
    this.getTextList()
  }
}
