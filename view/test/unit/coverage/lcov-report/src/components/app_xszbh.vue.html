import common from '../common.js'
export default {
  data () {
    return {
      selfStatus: null,
      selfMsg: null,
      applyReson: null,
      isApply: false,
      geted: false,
      geted2: false
    }
  },
  methods: {
    doneThing: function (data) {
      console.log(data)
      if (data.status) {
        this.getSelfStatus()
      };
      common.alertBox(data.message)
    },
    getSelfStatus () {
      common.ajax('get', '/info/idcards/status', this.setSelfStatus)
    },
    setSelfStatus (data) {
      console.log(data)
      if (data.status) {
        if (data.result && data.result.length > 0) {
          this.selfStatus = data.result[0]
          this.isApply = true
        }
        this.geted = true
      } else {
        common.alertBox(data.message)
      }
    },
    getSelfMsg () {
      this.geted2 = false
      common.ajax('get', '/info/idcards/user', this.setSelfMsg)
    },
    setSelfMsg (data) {
      console.log(data)
      if (data.status) {
        this.selfMsg = data.infos[0]
        this.geted2 = true
      }
    },
    submitApply () {
      let data = {
        apy_reson: this.applyReson
      }
      let jsonData = JSON.stringify(data)
      common.ajax('post', '/info/idcards/apply', this.doneThing, jsonData, 'json')
    }
  },
  ready () {
    this.getSelfStatus()
  }
}
