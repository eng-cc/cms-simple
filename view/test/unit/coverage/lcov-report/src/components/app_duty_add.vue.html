        activityType: null,
        activityTime: null,
        activityTime2: null,
        activityWeekday: null,
        activityXiaoqu: null

      timeMsg: {
        activityDepartment: null,
        departmentPeoples: null,
        activityPeople: null,
        activityType: null,
        activityTime: null,
        activityTime2: null,
        activityXiaoqu: null
      },
      weekdayList: null,
      userDepartment: localStorage.getItem('userDepartantId'),
      xiaoqu: JSON.parse(localStorage.getItem('xiaoqu')),
      geted: 0
    }
  },
  methods: {
    doneThing: function (data) {
      console.log(data)
      // if (data.status) {
      //   // this.getApplyList()
      // };
      common.alertBox(data.message)
    },
    getDate () {
      let date = new Date()
      this.activityTime = this.activityTime2 = date.getFullYear() + '-' + (date.getMonth() + 1) + '-' + date.getDate()
    },
    getworkPlace () {
      common.ajax('get', '/duty/department', this.setworkPlace)
    },
    setworkPlace (data) {
      console.log(data)
      if (data.status) {
        this.workPlace = data.contactsc
        this.geted = this.geted + 1
      } else {
        common.alertBox(data.message)
      }
    },
    getDepartmentPeoples (obj) {
      this.departmentPeoples = null
      common.ajax('get', '/duty/teachers/' + obj.activityDepartment, this.setDepartmentPeoples)
    },
    setDepartmentPeoples (data) {
      console.log(data)
      if (data.status) {
        this.departmentPeoples = data.contacts
      } else {
        common.alertBox(data.message)
      }
    },
    getTypeList () {
      common.ajax('get', '/duty/type', this.setTypeList)
    },
    setTypeList (data) {
      console.log(data)
      if (data.status) {
        this.typeList = data.result
        this.geted = this.geted + 1
      } else {
        common.alertBox(data.message)
      }
    },
    getWeekdayList () {
      common.ajax('get', '/duty/weekday', this.setWeekdayList)
    },
    setWeekdayList (data) {
      console.log(data)
      if (data.status) {
        this.weekdayList = data.result
        this.geted = this.geted + 1
      } else {
        common.alertBox(data.message)
      }
    },
    addByDay () {
      let data = {
        dw_code: this.dayMsg.activityDepartment,
        zgh_code: this.dayMsg.activityPeople,
        zblx_code: this.dayMsg.activityType,
        zbrq: this.dayMsg.activityTime,
        zbdd_code: this.dayMsg.activityXiaoqu
      }
      console.log(data)
      let jsonData = JSON.stringify(data)
      common.ajax('post', '/duty/arrange/day', this.doneThing, jsonData, 'json')
    },
    addByWeekday () {
      let data = {
        dw_code: this.weekdayMsg.activityDepartment,
        zgh_code: this.weekdayMsg.activityPeople,
        zblx_code: this.weekdayMsg.activityType,
        bzbrq: this.weekdayMsg.activityTime,
        ezbrq: this.weekdayMsg.activityTime2,
        week_code: this.weekdayMsg.activityWeekday,
        zbdd_code: this.weekdayMsg.activityXiaoqu
      }
      console.log(data)
      let jsonData = JSON.stringify(data)
      common.ajax('post', '/duty/arrange/week', this.doneThing, jsonData, 'json')
    },
    addByTime () {
      let data = {
        dw_code: this.timeMsg.activityDepartment,
        zgh_code: this.timeMsg.activityPeople,
        zblx_code: this.timeMsg.activityType,
        bzbrq: this.timeMsg.activityTime,
        ezbrq: this.timeMsg.activityTime2,
        zbdd_code: this.timeMsg.activityXiaoqu
      }
      console.log(data)
      let jsonData = JSON.stringify(data)
      common.ajax('post', '/duty/arrange/time', this.doneThing, jsonData, 'json')
    }
  },
  ready () {
    common.datepickerInit()

    this.getworkPlace()
